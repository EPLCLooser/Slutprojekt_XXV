get('/events') do 
  type_search = params[:event_type]
  if type_search == ""
    type_search = nil
  end
  if is_admin?()
    @admin = true
    @users = get_users()
  end
  @owned_events, @joined_events = show_events(type_search)
  slim(:index)
end

  date = params[:date]
  time = params[:time]
